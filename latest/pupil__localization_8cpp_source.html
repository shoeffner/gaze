62 void fill_displacement_tables(
63  dlib::matrix<double>& table_x, // NOLINT
64  dlib::matrix<double>& table_y, // NOLINT
65  int size) {
66  // Ensure size is bigger than the current size (i.e. if an update is needed
67  // at all)
68  if (size <= table_x.nr()) {
69  return;
70  }
71 
72  // Ensure size is odd
73  size += (1 - size & 1);
74  int half_size = (size - 1) / 2;
75 
76  // Resize tables
77  table_x.set_size(size, size);
78  table_y.set_size(size, size);
79 
80  // Fill with indices
81  for (int row = 0; row < table_x.nr(); ++row) {
82  dlib::set_rowm(table_x, row) = dlib::range(-half_size, half_size);
83  }
84  table_y = dlib::trans(table_x);
85 
86  // Calculate norms and store to table_x, then transpose to table_y
87  const double epsilon = std::numeric_limits<double>::min();
88  for (int row = 0; row < table_x.nr(); ++row) {
89  for (int col = 0; col < table_x.nc(); ++col) {
90  table_x(row, col) = table_x(row, col) /
91  (std::sqrt(table_x(row, col) * table_x(row, col)
92  + table_y(row, col) * table_y(row, col))
93  // Avoid div0
94  + epsilon);
95  }
96  }
97  table_y = dlib::trans(table_x);
98 }

127 void PupilLocalization::process(util::Data& data) {
128  if (data.landmarks.num_parts() < 5) {
129  return;
130  }
131 
132  // Extract eye patches
133  std::vector<dlib::chip_details> details =
134  util::get_eyes_chip_details(data.landmarks);
135  dlib::extract_image_chips(data.image, details, data.eyes);
136 
137  // Resize lookup table if needed
138  int max_size = std::max({data.eyes[0].nr(), data.eyes[0].nc(),
139  data.eyes[1].nr(), data.eyes[1].nc()});
140  if (2 * max_size > this->displacement_table_x.nr()) {
141  util::fill_displacement_tables(
142  this->displacement_table_x,
143  this->displacement_table_y,
144  2 * max_size + 7);
145  }
146 
147  for (int i = 0; i < 2; ++i) {
148  dlib::matrix<double> eye_in;
149  dlib::assign_image(eye_in, data.eyes[i]);
150 
151  // Calculate gradients
152  dlib::matrix<double> eye_horizontal;
153  dlib::matrix<double> eye_vertical;
154  dlib::sobel_edge_detector(eye_in, eye_horizontal, eye_vertical);
155 
156  util::normalize_and_threshold_gradients(eye_horizontal, eye_vertical,
157  this->relative_threshold_factor);
158 
159  // Blur eyes as weights
160  int boundary_offset = 6;
161  dlib::matrix<double> eye_gaussian;
162  dlib::gaussian_blur(eye_in, eye_gaussian, this->sigma, boundary_offset);
163  eye_gaussian = 255 - eye_gaussian;
164 
165  // Compute objective function t and select highest value
166  double max_t = std::numeric_limits<double>::min();
167  int argmax_r;
168  int argmax_c;
169 
170  int center = (this->displacement_table_x.nr() - 1) / 2;
171  auto nr = eye_in.nr();
172  auto nc = eye_in.nc();
173  dlib::matrix<double> zero = dlib::zeros_matrix<double>(nr, nc);
174  dlib::matrix<double> ts = dlib::zeros_matrix<double>(nr, nc);
175  for (auto row = decltype(nr){boundary_offset};
176  row < nr - boundary_offset; ++row) {
177  for (auto col = decltype(nc){boundary_offset};
178  col < nc - boundary_offset; ++col) {
179  dlib::matrix<double> d_x = dlib::subm(this->displacement_table_x,
180  center - row, center - col, nr, nc);
181  dlib::matrix<double> d_y = dlib::subm(this->displacement_table_y,
182  center - row, center - col, nr, nc);
183 
184  double t =
185  dlib::sum(
186  dlib::squared(dlib::max_pointwise(
187  dlib::pointwise_multiply(d_x, eye_horizontal) +
188  dlib::pointwise_multiply(d_y, eye_vertical), zero))) *
189  eye_gaussian(row, col);
190  if (t > max_t) {
191  max_t = t;
192  argmax_r = row;
193  argmax_c = col;
194  }
195  }
196  }
197  data.centers[i] = dlib::point(argmax_c, argmax_r);
198  }
