43 void HeadPoseEstimation::update_overlay(const util::Data& data) {
44  this->widget->clear_overlay();
45  if (data.landmarks.num_parts() <= 0) {
46  return;
47  }
48 
49  // Project reference points to image.
50  std::vector<cv::Point3d> base_vecs =
51  {{0, 0, 0},
52  {0.1 * this->model_scale, 0, 0},
53  {0, 0.1 * this->model_scale, 0},
54  {0, 0, 0.1 * this->model_scale}};
55  std::vector<cv::Point2d> image_points;
56  cv::projectPoints(base_vecs, data.head_rotation, data.head_translation,
57  this->read_or_set_camera_matrix(data),
58  this->get_and_maybe_read_distortions(data), image_points);
59 
60  // coordinate system
61  for (auto i = decltype(this->coord_base_overlay.size()){0};
62  i < this->coord_base_overlay.size(); ++i) {
63  this->coord_base_overlay[i] = dlib::image_display::overlay_line(
64  dlib::point(image_points[0].x, image_points[0].y),
65  dlib::point(image_points[i + 1].x, image_points[i + 1].y),
66  this->coord_base_overlay[i].color);
67  }
68  this->widget->add_overlay(this->coord_base_overlay);
69 
70  // detected vs. projected points in overlay
71  std::vector<cv::Point2d> projected_points;
72  cv::projectPoints(this->model_points, data.head_rotation,
73  data.head_translation, this->read_or_set_camera_matrix(data),
74  this->get_and_maybe_read_distortions(data), projected_points);
75 
76  auto color_detection = dlib::rgb_pixel(255, 0, 0);
77  auto color_projection = dlib::rgb_pixel(0, 0, 255);
78  std::vector<dlib::image_display::overlay_circle> detections;
79  std::vector<dlib::image_display::overlay_circle> projections;
80  for (auto i = decltype(projected_points.size()){0};
81  i < projected_points.size(); ++i) {
82  detections.push_back(dlib::image_display::overlay_circle(
83  data.landmarks.part(this->index_list[i]), 2, color_detection,
84  "d" + std::to_string(this->index_list[i])));
85  projections.push_back(dlib::image_display::overlay_circle(
86  dlib::point(projected_points[i].x, projected_points[i].y),
87  2, color_projection, "p" + std::to_string(this->index_list[i])));
88  }
89  this->widget->add_overlay(detections);
90  this->widget->add_overlay(projections);
91 }
