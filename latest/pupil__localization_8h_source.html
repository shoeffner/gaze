17 /**
18  * Filles the displacement table to precalculate values for vector @f$d_i@f$
19  * @cite Timm2011.
20  *
21  * The function returns without modifying table_x or table_y if size is
22  * smaller than the current table size.
23  *
24  * The size will always be set to size or size + 1, whichever is odd. This
25  * ensures that the center value is 0 as a reference point.
26  *
27  * After a call to this function with a sufficiently big size argument,
28  * table_x and table_y contain values such that reading from both tables
29  * at the same indices yields the respective components for a unit vector
30  * geometrically pointing from the center of the matrix to the read position.
31  *
32  * A @f$3 \times 3@f$ table_x would thus look like this:
33  *
34  * @f[
35  * \left(
36  * \begin{array}{ccc}
37  * -0.707 & 0 & 0.707 \\
38  * -1 & 0 & 1 \\
39  * -0.707 & 0 & 0.707
40  * \end{array}
41  * \right)
42  * @f]
43  *
44  * And the corresponding @f$3 \times 3@f$ table_y would be table_x's transposed
45  * version:
46  *
47  * @f[
48  * \left(
49  * \begin{array}{ccc}
50  * -0.707 & -1 & -0.707 \\
51  * 0 & 0 & 0 \\
52  * 0.707 & 1 & 0.707
53  * \end{array}
54  * \right)
55  * @f]
56  *
57  * Accessing table_x and table_y at (0, 0) yields @f$(-0.707, -0.707)@f$ which
58  * is a unit length vector pointing from the center (1, 1) to (0, 0).
59  *
60  * @param table_x The x components of @f$d_i@f$.
61  * @param table_y The y components of @f$d_i@f$.
62  * @param size The size to grow this table to.
63  */
64 void fill_displacement_tables(dlib::matrix<double>& table_x, // NOLINT
65  dlib::matrix<double>& table_y, // NOLINT
66  int size);

68 /**
69  * Normalizes a horizontal and a vertical gradient matrix to unit length vectors.
70  *
71  * If a relative threshold is provided, all values where the gradient magnitude
72  * @f$ M @f$ is lower than @f$ \theta \max M @f$ are set to 0.
73  *
74  * @param horizontal A gradient, e.g. from dlib::sobel_edge_detector
75  * @param vertical A gradient, e.g. from dlib::sobel_edge_detector
76  * @param relative_threshold optional threshold to set small values to 0.
77  */
78 template <typename T>
79 void normalize_and_threshold_gradients(
80  dlib::matrix<T>& horizontal, // NOLINT
81  dlib::matrix<T>& vertical, // NOLINT
82  double relative_threshold = -1) {
83  dlib::matrix<T> magnitude;
84  magnitude = dlib::sqrt(dlib::squared(horizontal) +
85  dlib::squared(vertical));
86 
87  // Thresholding
88  if (relative_threshold >= 0) {
89  T threshold = dlib::mean(magnitude)
90  + dlib::stddev(magnitude) * relative_threshold;
91  for (int row = 0; row < horizontal.nr(); ++row) {
92  for (int col = 0; col < horizontal.nc(); ++col) {
93  if (magnitude(row, col) < threshold) {
94  horizontal(row, col) = 0;
95  vertical(row, col) = 0;
96  }
97  }
98  }
99  }
100 
101  // Normalization
102  horizontal = dlib::pointwise_multiply(
103  horizontal, dlib::reciprocal(magnitude));
104  vertical = dlib::pointwise_multiply(
105  vertical, dlib::reciprocal(magnitude));
106 }
