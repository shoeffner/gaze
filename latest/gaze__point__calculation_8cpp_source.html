20 double parse_aspect_ratio(std::string aspect_ratio_string) {
21  double aspect_ratio;
22  int delim = aspect_ratio_string.find(":");
23  if (delim < 0) {
24  delim = aspect_ratio_string.find("/");
25  }
26  if (delim > 0) {
27  double width_rate = std::stod(aspect_ratio_string.substr(0, delim));
28  double height_rate = std::stod(aspect_ratio_string.substr(delim + 1));
29  aspect_ratio = width_rate / height_rate;
30  } else {
31  aspect_ratio = std::stod(aspect_ratio_string);
32  }
33  return aspect_ratio;
34 }

136 std::vector<cv::Vec3d> GazePointCalculation::unprojectPoints(
137  const std::vector<cv::Vec2d>& points, const cv::Vec3d& translation,
138  const cv::Matx33d& rotation, double distance) {
139  // Half turn about up-vector to account for camera mirror image vs. model
140  cv::Matx33d rcam;
141  rcam << -1, 0, 0,
142  0, 1, 0,
143  0, 0, -1;
144  std::vector<cv::Vec2d> restored_points;
145  cv::undistortPoints(points, restored_points,
146  this->camera_matrix, this->distortion_coefficients);
147  std::vector<cv::Vec3d> unprojectedPoints;
148  for (const cv::Vec2d& point : restored_points) {
149  cv::Vec3d unprojected(point[0], point[1], 1);
150  unprojectedPoints.push_back(
151  rcam * rotation * ((unprojected / distance) - translation));
152  }
153  return unprojectedPoints;
154 }

166 std::vector<cv::Vec3d> GazePointCalculation::get_screen_corners(
167  const cv::Vec3d& camera_pos, const cv::Matx33d& rotation) {
168  cv::Vec3d screen_tl =
169  cv::Vec3d(-this->camera_offset_x, -this->camera_offset_y, 0);
170  cv::Vec3d screen_tr = screen_tl + cv::Vec3d(this->screen_width_m, 0, 0);
171  cv::Vec3d screen_br = screen_tr + cv::Vec3d(0, this->screen_height_m, 0);
172  cv::Vec3d screen_bl = screen_br + cv::Vec3d(-this->screen_width_m, 0, 0);
173  auto transform = [&rotation, &camera_pos] (cv::Vec3d in) -> cv::Vec3d {
174  return rotation.t() * in + camera_pos;
175  };
176  return {transform(screen_tl), transform(screen_tr),
177  transform(screen_br), transform(screen_bl)};
178 }

180 cv::Matx32d GazePointCalculation::calculate_gaze_point(
181  const cv::Vec3d& eye_ball_center, const cv::Vec3d& pupil,
182  const cv::Vec3d& screen_a, const cv::Vec3d& screen_b,
183  const cv::Vec3d& screen_c) {
184  cv::Matx33d raycast;
185  cv::Mat components[] = {cv::Mat(eye_ball_center - pupil),
186  cv::Mat(screen_b - screen_a),
187  cv::Mat(screen_c - screen_a)};
188  cv::hconcat(components, 3, raycast);
189  cv::Vec3d tuv = (raycast.inv() * (eye_ball_center - screen_a));
190  cv::Vec3d gaze_point = eye_ball_center + (pupil - eye_ball_center) * tuv[0];
191 
192  cv::Matx32d result;
193  cv::Mat vecs[] = {cv::Mat(gaze_point), cv::Mat(tuv)};
194  cv::hconcat(vecs, 2, result);
195  return result;
196 }

206 void GazePointCalculation::process(util::Data& data) {
207  if (data.landmarks.num_parts() < 5) {
208  return;
209  }
210 
211  // Convert landmarks to OpenCV data structures
212  std::vector<cv::Vec3d> landmarks;
213  for (auto i : this->landmark_indices) {
214  cv::Vec3d landmark(data.landmarks.part(i).x(), data.landmarks.part(i).y(),
215  0);
216  landmarks.push_back(landmark);
217  }
218 
219  // Project pupils into model
220  cv::Matx34d transformation = this->invertProjection(landmarks);
221 
222  std::vector<dlib::chip_details> eyes =
223  util::get_eyes_chip_details(data.landmarks);
224 
225  for (auto i = decltype(data.pupils.size()){0};
226  i < data.pupils.size(); ++i) {
227  dlib::point eye = data.centers[i] + eyes[i].rect.tl_corner();
228  data.pupils[i] = transformation * cv::Vec4d(eye.x(), eye.y(), 0, 1);
229  }
230 
231  // Calculate distance
232  double distance = this->calculate_distance(
233  data.landmarks.part(36), data.landmarks.part(45), 0.08671);
234  cv::Matx33d R;
235  cv::Rodrigues(data.head_rotation, R);
236 
237  cv::Vec3d model_to_camera_dir =
238  this->get_model_to_camera_dir(data.head_translation, R);
239  cv::Vec3d camera_pos = this->get_camera_pos(model_to_camera_dir, distance);
240 
241  std::vector<cv::Vec3d> screen_tl_tr_br_bl =
242  this->get_screen_corners(camera_pos, R);
243 
244  // Ray cast
245  cv::Vec2d screen_coord_coeffs(0, 0);
246  for (auto i = decltype(data.gaze_points.size()){0};
247  i < data.gaze_points.size(); ++i) {
248  cv::Matx32d gaze_point = this->calculate_gaze_point(
249  this->eye_ball_centers[i], data.pupils[i],
250  screen_tl_tr_br_bl[0], screen_tl_tr_br_bl[1], screen_tl_tr_br_bl[3]);
251  data.gaze_points[i] =
252  cv::Vec3d(gaze_point(0, 0), gaze_point(1, 0), gaze_point(2, 0));
253  screen_coord_coeffs[0] += gaze_point(1, 1);
254  screen_coord_coeffs[1] += gaze_point(1, 2);
255  }
256  screen_coord_coeffs /= 2;
257  screen_coord_coeffs[0] = util::clamp(screen_coord_coeffs[0], 0, 1);
258  screen_coord_coeffs[1] = util::clamp(screen_coord_coeffs[1], 0, 1);
259 
260  data.estimated_gaze_point = cv::Vec2d(
261  this->screen_width_px * screen_coord_coeffs[0],
262  this->screen_height_px * screen_coord_coeffs[1]);
263 }
