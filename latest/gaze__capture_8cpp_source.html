87  dlib::set_subm(mask_in, data.landmarks.get_rect()) = 1;
88  dlib::matrix<unsigned char, 25, 25> mask_out;
89  dlib::resize_image(mask_in, mask_out, dlib::interpolate_nearest_neighbor());
90  cv::Mat mask = dlib::toMat(mask_out);
91 
92  cv::Vec2d prediction_cam = this->itracker->predict(left, right, face, mask);
93  // Note that the prediction is in cm, thus DIV100 is needed
94  double x = (prediction_cam[0] / 100 + this->camera_offset[0])
95  / this->screen_size_m[0] * this->target_size[0];
96  double y = (-prediction_cam[1] / 100 + this->camera_offset[1])
97  / this->screen_size_m[1] * this->target_size[1];
98  data.estimated_gaze_point = cv::Vec2d(x, y);
