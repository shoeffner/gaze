57 cv::Mat compute_x_gradient(const cv::Mat& image) {

81 cv::Mat matrix_magnitude(const cv::Mat& mat_x, const cv::Mat& mat_y) {
82  cv::Mat mags(mat_x.rows, mat_x.cols, CV_64F);
83  for (auto y = decltype(mat_x.rows){0}; y < mat_x.rows; ++y) {
84  const double *Xr = mat_x.ptr<double>(y), *Yr = mat_y.ptr<double>(y);
85  double *row = mags.ptr<double>(y);

111 void test_possible_centers_formula(
112  int x, int y,
113  const cv::Mat& weight,
114  double gx, double gy,
115  cv::Mat& out) { // NOLINT
116  for (auto cy = decltype(out.rows){0}; cy < out.rows; ++cy) {
117  double *out_row = out.ptr<double>(cy);
118  const unsigned char *Wr = weight.ptr<unsigned char>(cy);
119  for (auto cx = decltype(out.cols){0}; cx < out.cols; ++cx) {
120  if (x == cx && y == cy) {
121  continue;
122  }
123  double dx = x - cx;
124  double dy = y - cy;
125  // normalize d
126  double magnitude = std::sqrt(dx * dx + dy * dy);
127  dx = dx / magnitude;
128  dy = dy / magnitude;
129  double dotProduct = dx * gx + dy * gy;
130  dotProduct = std::max(0.0, dotProduct);
131  out_row[cx] += dotProduct * dotProduct * Wr[cx];
132  }
133  }
134 }

165 cv::Point find_eye_center(
166  const cv::Mat& face, const cv::Rect& eye,
167  double gradient_threshold, double fast_eye_width = 50) {
168  cv::Mat eye_roi_unscaled = face(eye);
169  cv::Mat eye_roi;
170  cv::resize(
171  eye_roi_unscaled,
172  eye_roi,
173  cv::Size(fast_eye_width,
174  fast_eye_width / eye_roi_unscaled.cols * eye_roi_unscaled.rows));
175 
176  cv::Mat gradientX = compute_x_gradient(eye_roi);
177  cv::Mat gradientY = compute_x_gradient(eye_roi.t()).t();
178  cv::Mat mags = matrix_magnitude(gradientX, gradientY);
179  double gradientThresh = compute_dynamic_threshold(mags,
180  gradient_threshold);
181 
182  for (auto y = decltype(eye_roi.rows){0}; y < eye_roi.rows; ++y) {
183  double *x_row = gradientX.ptr<double>(y);
184  double *y_row = gradientY.ptr<double>(y);
185  const double *M_ROW = mags.ptr<double>(y);
186  for (auto x = decltype(eye_roi.cols){0}; x < eye_roi.cols; ++x) {
187  if (M_ROW[x] > gradientThresh) {
188  x_row[x] /= M_ROW[x];
189  y_row[x] /= M_ROW[x];
190  } else {
191  x_row[x] = 0.0;
192  y_row[x] = 0.0;
193  }
194  }
195  }
196 
197  cv::Mat weight;
198  cv::GaussianBlur(eye_roi, weight, cv::Size(5, 5), 0, 0);
199  weight = 255 - weight;
200 
201  cv::Mat outSum = cv::Mat::zeros(eye_roi.rows, eye_roi.cols, CV_64F);
202  for (int y = 0; y < weight.rows; ++y) {
203  const double *X_ROW = gradientX.ptr<double>(y);
204  const double *Y_ROW = gradientY.ptr<double>(y);
205  for (int x = 0; x < weight.cols; ++x) {
206  if (X_ROW[x] == 0.0 && Y_ROW[x] == 0.0) {
207  continue;
208  }
209  test_possible_centers_formula(x, y, weight, X_ROW[x], Y_ROW[x], outSum);
210  }
211  }
212 
213  cv::Point maxP;
214  double maxVal;
215  cv::minMaxLoc(outSum, nullptr, &maxVal, nullptr, &maxP);
216 
217  return unscale_point(maxP, eye, fast_eye_width);
218 }
